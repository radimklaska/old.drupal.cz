Formulář v tabulce

Vložil/a Anonymous, 30 Červenec, 2009 - 10:46
Kategorie: Jak na to?
Týká se verze: Drupal 6.x

Ahoj, pročetl jsem si několik článků o tomto tématu, ale žádnej jednoznačně nedefinuje, co má všechno (jaké hook funkce) musí obsahovat funkce, které zapříčiní, že se mi formulář zobrazí pěkně v tabulce, jako když třeba kouknu do odkazu moduly.

Mám tento kód a nevím co je špatně, formulář se stále zobrazuje „standardně“, tedy ne jak v tabulce:

function example_service_page() {
    $out = drupal_get_form('example_include_form');
    return $out;
}

function example_include_form() {

    $query = db_query("SELECT * FROM {example_table}");

    if(mysqli_num_rows($query)>0) {

        $id = 1;
        while ($row = db_fetch_object($query)) {

            $form['ex'][$id] = array(
                '#type' => 'fieldset',
                '#title' => t(utf8('Editace služeb')),
                '#collapsible' => FALSE,
                '#tree' => TRUE,
            );

            $form['ex'][$id]['service-name'] = array(
                '#type' => 'textfield',
                '#size' => 80,
                '#title' => t(utf8('Název')),
                '#maxlength' => 355,
                '#default_value' => check_plain($row->name),
            );

            $form['ex'][$id]['root-service'] = array(
                '#type' => 'select',
                '#title' => t(utf8('Nadřazená služba')),
                '#options' => get_root_services(true),
                '#default_value' => check_plain($row->root_s),
            );

            $id++;

        }

        $form['ex']['change'] = array(
            '#type' => 'submit',
            '#value' => t(utf8('Edituj službu')),
            '#submit' => array('ex_update'),
        );

        $form['ex']['delete'] = array(
            '#type' => 'submit',
            '#value' => t(utf8('Smaž službu')),
            '#submit' => array('ex_delete'),
        );

        $form['ex']['all-check'] = array(
            '#type' => 'item',
            '#value' => '<a href="#checkAll" id="checkAll">'.utf8('Vyber všechny').'</a> | <a href="#uncheckAll" id="uncheckAll">'.utf8('Odznač všechny').'</a>',
        );


        return $form;

    }
}

function theme_example_include_form($form) {

    $output = '';

    // nastavime si hlavicku tabulky
    $header = array(
        array('data' => 'Nazev'),
        array('data' => 'Root'),
    );


    foreach (element_children($form['ex']) as $key) {

        if (is_array($element) && element_child($key)) {

        // riadok
            $rows[] = array(
                'data' => array(
                array('data' => $key.' '.drupal_render($form['ex'][$key])),
                array('data' => $key.' '.drupal_render($form['ex'][$key])),
                ),
            );

        }
    }

    $rows[] = array(
        array('data' => drupal_render($form['change'])),
        array('data' => drupal_render($form['delete'])),
    );

    $output .= theme('table', $header, $rows);
    $output .= drupal_render($form);

    return $output;

}

function edit_delete_transfer_service_include_form() {

    $query = db_query("SELECT * FROM {nejaka_table}");

        $form['edit_delete_transfer_service_include']['overview'] = array(

            $form['edit_delete_transfer_service_include']['overview']['rows'][$row->id]['service_id'] = array(
                '#value' => $row->id,
            );

            $form['edit_delete_transfer_service_include']['overview']['rows'][$row->id]['service-name'] = array(
                '#type' => 'textfield',
                '#size' => 80,
                '#title' => t(utf8('Název')),
                '#maxlength' => 355,
                '#default_value' => check_plain($row->name_of_s),
            );

            $form['edit_delete_transfer_service_include']['overview']['rows'][$row->id]['root-service'] = array(
                '#type' => 'select',
                '#title' => t(utf8('Nadřazená služba')),
                '#options' => get_root_services('transfer_service_services', true),
                '#default_value' => check_plain($row->root_s),
            );

            $form['edit_delete_transfer_service_include']['overview']['rows'][$row->id]['location'] = array(
                '#type' => 'radios',
                '#default_value' => 1,
                '#title' => t(utf8('Lokalita')),
                '#options' => array(
                1 => t('po Praze'),
                2 => t('mimo Prahu'),
                ),
                '#default_value' => check_plain($row->location),
            );

            $form['edit_delete_transfer_service_include']['overview']['rows'][$row->id]['edit'] = array(
                '#value' => implode(", ", array(
                l(t('Edituj'), "transfer_service/{$row->id}/edit"),
                ))
            );
        }
        return $form;

    }
}


function theme_edit_delete_transfer_service_include_form($form) {

    $header = array('service_id', t('Nazev'), t('Koren'), t('Lokalita'), t(''));
    $rows = array();

    foreach (element_children($form['rows']) as $key) {
        $row = array();
        $row[] = drupal_render($form['rows'][$key]['service_id']);
        $row[] = drupal_render($form['rows'][$key]['service-name']);
        $row[] = drupal_render($form['rows'][$key]['root-service']);
        $row[] = drupal_render($form['rows'][$key]['location']);
        $row[] = drupal_render($form['rows'][$key]['edit']);
        $rows[] = $row;
    }

    $row = array();
    $row[] = drupal_render($form['rows']['service_id']);
    $row[] = drupal_render($form['rows']['service-name']);
    $row[] = '';
    $rows[] = $row;

    $output = theme('table', $header, $rows);
    return $output;
}
