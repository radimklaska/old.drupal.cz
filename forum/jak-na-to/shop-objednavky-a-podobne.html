Mno – Drupal 5 ma typy

Napsal uživatel Havran dne 17 Leden, 2007 - 19:03.

Mno – Drupal 5 ma typy obsahu – urobil by som si vlastny typ obsahu (trebars pod nazvom tovar) a ten by som pomocou CCK rozsiril o parametre daneho tovaru (alebo co to ma byt) – cena, hmotnost atd. Co sa kosiku tyka to je kusok zlozitejsie – ja som si to naprogramoval sam na Drupal 4.7 – v podstate som postupoval podobne s tym ze som si vytvoril node typu kniha a nakupny kosik som naprogramoval ako blok.

Cele je to vidiet na www.enigma.sk a bola to moja prva Drupal stranka… Len lutujem ze som vtedy nemal uz Drupal 5 :) CCK som tam vobec nepouzil (tusim ho este len dokopy davali, vlastne aj drupal bol 4.7 beta).

<?php
// $Id$

/**
 * @file
 * DRUPAL MODULE - Enigma - bookstore - shoping cart (block)
 */
/**
 * Implementation of hook_help().
 */
function enigma_shoping_cart_help($section) {
  switch ($section) {
    case 'admin/help#enigma_shoping_cart':
      $output = '<p>ENIGMA - nákupný košík</p>';
      return $output;
    case 'admin/modules#description':
      return 'ENIGMA - nákupný košík.';
  }
}

/**
 * bloky pre zobrazenie zoznamu kategorii a zoznamu knih v kategorii
 */
function enigma_shoping_cart_block($op = 'list', $delta = 0, $edit = array()) {
  // The $op parameter determines what piece of information is being requested.
  switch ($op) {
    case 'list':
      // If $op is "list", we just need to return a list of block descriptions.
      // This is used to provide a list of possible blocks to the administrator,
      // end users will not see these descriptions.
      $blocks[0]['info'] = 'ENIGMA: nákupný košík';
      return $blocks;
    case 'configure':
      // If $op is "configure", we need to provide the administrator with a
      // configuration form. The $delta parameter tells us which block is being
      // configured. In this example, we'll allow the administrator to customize
      // the text of the first block.
      $form = array();
      return $form;
    case 'save':
      // If $op is "save", we need to save settings from the configuration form.
      // Since the first block is the only one that allows configuration, we
      // need to check $delta to make sure we only save it.
      return;
    case 'view': default:
      // If $op is "view", then we need to generate the block for display
      // purposes. The $delta parameter tells us which block is being requested.
      switch ($delta) {
        case 0:
          $block['subject'] = 'Nákupný košík';
          $block['content'] = enigma_shoping_cart_block_render($block);
          break;
      }
      return $block;
  }
}

// zobrazime blok z kategoriami knih
function enigma_shoping_cart_block_render(&$block) {
  $output = '';

  // toto vsetko robime len ked nie je kosik prazdny
  if (!empty($_SESSION['nakupny_kosik']['obsah'])) {
    // spristupnime si typ prihlaseneho pouzivatela
    global $user;
    $profile_type = _get_profile_type($user);

    // manipulacia z kosikom
    if (!empty($_POST) && $_POST['edit']['form_id'] == 'nakupny_kosik') {
      // vymazeme vybrane polozky
      if (!empty($_POST['delete_kupa'])) {
        foreach ($_POST['delete_kupa'] as $key => $val) {
          unset($_SESSION['nakupny_kosik']['obsah'][$key]);
        }
      }
      // upravime obsah nakupneho kosika
      foreach ($_SESSION['nakupny_kosik']['obsah'] as $key => $val) {
        if ((int)$_POST['pocet_kupa'][$key] == 0) unset($_SESSION['nakupny_kosik']['obsah'][$key]);
        else $_SESSION['nakupny_kosik']['obsah'][$key] = (int)$_POST['pocet_kupa'][$key];
      }

      if (!empty($_POST['ukoncit_nakup'])) {
        unset($_SESSION['form_ok']);
        unset($_SESSION['objednavka_ok']);
        drupal_goto('order-list/'.$profile_type);
      }
      else {
        // presmerujeme kvoli moznemu znovuodoslaniu post
        drupal_goto($_GET['q']);
      }
    }

    // pocet objednanych knih
    $pocet_objednanych_knih = array_sum((array)$_SESSION['nakupny_kosik']['obsah']);

    // sucty v hlavicke
    switch ($pocet_objednanych_knih)
    {
      case 0: return; // pokial uz nezostava v kosiku ziadna kniha skoncime
      case 1: $pocet_objednanych_knih_str = ' vloženú '.$pocet_objednanych_knih.' knihu'; break;
      case 2:
      case 3:
      case 4: $pocet_objednanych_knih_str = ' vložené '.$pocet_objednanych_knih.' knihy'; break;
      default: $pocet_objednanych_knih_str = ' vložených '.$pocet_objednanych_knih.' kníh';
    }

    // kosik otvor zatvor
    if (empty($_COOKIE['bin_status']) || $_COOKIE['bin_status'] == '') {
      $_COOKIE['bin_status'] = true;
      setcookie('bin_status', 'true');
    }
    if ($_COOKIE['bin_status'] == 'true') {
      $nakupny_kosik_class = 'nakupnyKosikClass';
      $kosik_otvor_zatvor = 'kosik-zatvor';
      $kosik_otvor_zatvor_str = 'Zatvor nákupný košík';
    } else {
      $nakupny_kosik_class = 'nakupnyKosikClassH';
      $kosik_otvor_zatvor = 'kosik-otvor';
      $kosik_otvor_zatvor_str = 'Otvor nákupný košík';
    }

    // nastavenia podla typu pouzivatela
    switch ($profile_type) {
      case PROFILE_TYP_KNIHKUPEC:
        $info_str = 'Zobrazené ceny a rabaty zodpovedajú cenám pre prihláseného kníhkupca.';
        $colspan_num_head = 4;
        $rabat_header = true;
        break;
      case PROFILE_TYP_SPOLUPRACOVNIK:
        $info_str = 'Zobrazené ceny zodpovedajú cenám pre prihláseného spolupracovníka.';
        $colspan_num_head = 5;
        $rabat_header = false;
        break;
      case PROFILE_TYP_POUZIVATEL:
      default:
        $info_str = 'Zobrazené ceny zodpovedajú cenám pre koncového zákazníka. Pokiaľ sa prihlásite ako kníhkupec ceny budú automaticky prepočítané. Knihy na ktoré neposkytujeme rabat, budú po prihlásení kníhkupca automaticky odstránené z košíka!';
        $colspan_num_head = 5;
        $rabat_header = false;
    }

    // vyberieme udaje knih z db
    $sql = "SELECT
              no.nid,no.type,no.title,
              ek.evidc,ek.hmotnost,ek.autor,ek.cena_koncovy,ek.zlava,ek.cena_knihkupci,ek.rabat
            FROM {node} AS no
            LEFT JOIN {node_revisions} AS nr ON no.nid=nr.nid
            LEFT JOIN {enigma_knihy} AS ek ON nr.vid=ek.vid
            LEFT JOIN {term_node} AS tn ON tn.nid=no.nid
            WHERE no.type='enigma_kniha' AND no.vid=nr.vid AND (ek.nid=%s) AND status<>0
            ORDER BY ek.meno";
    $res = db_query($sql,implode(' OR ek.nid=',array_keys($_SESSION['nakupny_kosik']['obsah'])));

    $count = 1;
    $suma_hmotnost = 0;
    $cena_celkom = 0;
    $cart_item = array();
    while ($item = db_fetch_object($res)) {
      // musime zistit cenu podla prihlaseneho pouzivatela
      switch ($profile_type) {
        case PROFILE_TYP_KNIHKUPEC:
          $cena = $item->cena_knihkupci == 0 || empty($item->cena_knihkupci) ? FALSE : $item->cena_knihkupci;
          break;
        case PROFILE_TYP_SPOLUPRACOVNIK:
        case PROFILE_TYP_POUZIVATEL:
        default:
          $cena = $item->cena_koncovy;
          if (trim($item->zlava) == '*') $zlava = (int)variable_get('ENIGMA_book_default_zlava', '10');
          else if (trim($item->zlava) == '0')  $zlava = 0;
          else  $zlava = (int)$item->zlava;
          $cena_zlava = $cena*((100-$zlava)/100);
      }

      if ($cena === FALSE) {
        unset($_SESSION['nakupny_kosik']['obsah'][$item->nid]);
      }
      else {
        // musime zistit ci mame zobrazit rabat
        // pokial ma knihkupec nastaveny rabat a kniha nema pouzije sa knihkupcov rabat
        if (!empty($user->profile_knihkupcov_rabat) && $item->rabat == 0) {
          $rabat = $user->profile_knihkupcov_rabat;
        }
        // inak pokial ma kniha nastaveny rabat pouzije sa ten
        else if ($item->rabat != 0) {
          $rabat = $item->rabat;
        }
        // inak sa pouzije default rabat
        else {
          $rabat = variable_get('ENIGMA_book_default_rabat', '26');
        }
        //$rabat = $item->rabat != 0 ? $item->rabat : variable_get('ENIGMA_book_default_rabat', '26');
        // vypocty celkom
        $cena_krat_kus = $_SESSION['nakupny_kosik']['obsah'][$item->nid] * $cena;
        $cena_krat_kus_zlava = $_SESSION['nakupny_kosik']['obsah'][$item->nid] * $cena_zlava;
        $cena_minus_rabat = $cena_krat_kus-($cena_krat_kus*($rabat/100));
        $suma_celkom += $cena_krat_kus;
        $suma_celkom_zlava += $cena_krat_kus_zlava;
        $suma_celkom_rabat += $cena_minus_rabat;
        $suma_hmotnost += $_SESSION['nakupny_kosik']['obsah'][$item->nid] * $item->hmotnost;
        // parne neparne riadky
        $flag = ($count % 2) ? FALSE : TRUE;
        // naplnime potrebne premenne
        $cart_item[$count]['nid'] = $item->nid;
        $cart_item[$count]['evidc'] = $item->evidc;
        $cart_item[$count]['cena_str'] = (is_whole_number($cena) ? $cena.',-' :  str_replace('.',',',sprintf("%01.2f",$cena)).'');
        $cart_item[$count]['zlava'] = $zlava;
        $cart_item[$count]['cena_zlava_str'] = (is_whole_number($cena_zlava) ? $cena_zlava.',-' :  str_replace('.',',',sprintf("%01.2f",$cena_zlava)).'');
        $cart_item[$count]['title'] = $item->title;
        $cart_item[$count]['hmotnost'] = $item->hmotnost;
        $cart_item[$count]['rabat_str'] = $rabat_header ? $rabat : '';
        //$cart_item[$count]['rabat_str'] = $rabat_header ? ($item->rabat != 0 ? $item->rabat : variable_get('ENIGMA_book_default_rabat', '26')) : '';
        $cart_item[$count]['row_style'] = ($flag ? 'o' : 'e');
      }
      $count++;
    }
    $suma_hmotnost = $suma_hmotnost;
    $suma_celkom_str = (is_whole_number($suma_celkom) ? $suma_celkom.',-' :  str_replace('.',',',sprintf("%01.2f",$suma_celkom)).'');
    $suma_celkom_zlava_str = (is_whole_number($suma_celkom_zlava) ? $suma_celkom_zlava.',-' :  str_replace('.',',',sprintf("%01.2f",$suma_celkom_zlava)).'');
    $suma_celkom_rabat_str = $rabat_header ? ((is_whole_number($suma_celkom_rabat) ? $suma_celkom_rabat.',-' :  str_replace('.',',',sprintf("%01.2f",$suma_celkom_rabat)).'')) : '';

    // vlozime sablonu ktoru naplnime s udajmi vytiahnutymi v predchadzajucom kode
    ob_start();
    include path_to_theme().'/block-enigma_shoping_cart.tpl.php';
    $output .= ob_get_contents();
    ob_end_clean();

    // na zaver pridame formular
    $form['kosik'] = array('#type' => 'markup', '#value' => $output);
    $output = drupal_get_form('nakupny_kosik', $form);
  }

  return $output;
}

/**
 * pomocne funkcie
 */

// zisti ci je cislo cele cislo
function is_whole_number($var)
{
  return (is_numeric($var) && (intval($var) == floatval($var)));
}
