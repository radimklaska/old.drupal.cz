/

* Set an HTTP response header for the current page.

*

* Note: when sending a Content-Type header, always include a ‚charset‘ type

* too. This is necessary to avoid security bugs (e.g. UTF-7 XSS).

*/
function drupal_set_he­ader($header = NULL) {

// We use an array to guarantee there are no leading or trailing delimiters.

// Otherwise, header('') could get called when serving the page later, which

// ends HTTP headers prematurely on some PHP versions.

static $stored_headers = array();


if (strlen($header)) {

header($header); //řádek 142 kde tedy podle mě vrací proměnou $header jelikož se něco nepovedlo, ale moc tomu nerozumim.

$stored_headers[] = $header;

}

return implode(„\n“, $stored_headers);
}

/

* Get the HTTP response headers for the current page.

*/
function drupal_get_he­aders() {

return drupal_set_he­ader();
