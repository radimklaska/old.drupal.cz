<?php
// $Id: texy.module,v 1.3.2.2 2006/09/22 19:43:45 havran Exp $

/**
 * @file
 * DRUPAL MODULE - Texy!
 */

/**
 * Implementation of hook_help().
 */
function texy_help($section) {
  switch ($section) {
    case 'admin/help#texy':
      $output = '<p>Texy! is text-to-XHTML formatter and converter library. It allows you to write structured documents
      without knowledge or using of HTML language.</p><p>You write documents in humane easy-to-read plain text format
      and Texy! converts it to structurally and valid XHTML code.</p><p>Texy! is one of the most complex formatters.
      Its possibilities covers images, links (anchors), nested lists, tables and has full support for CSS.</p><p>Usage examples:</p><ul><li><a
      href="http://texy.info/en">http://texy.info/en</a></li></ul>';
      return $output;
    case 'admin/modules#description':
      return 'Texy! module - text-to-XHTML formatter and converter library';
  }
}

/**
 * Implementation of hook_settings().

      with php code for Texy! implementation this highlighter. For example I use for syntax highlighting FSHL .

/**
 * Implementation of hook_filter().
 */
function texy_filter($op, $delta = 0, $format = -1, $text = '') {
  switch ($op) {
    case 'list':
      return array(0 => 'Texy! text-to-XHTML formatter and converter filter');

    case 'description':
      switch ($delta) {
        case 0:
          return 'Texy! is text-to-XHTML formatter and converter library. It allows you to write structured
                  documents without knowledge or using of HTML language.';
        default:
          return;
      }
      break;

    case 'prepare':
      switch ($delta) {
        case 0:
          return $text;
        default:
          return $text;
        }
      break;

    case 'process':
      switch ($delta) {
        case 0:
          return _filter_texy($text);
        default:
          return $text;
      }
      break;

    case 'settings':
      switch ($delta) {
        default:
          return;
      }
      break;

    default:
      return $text;
  }
}

/**
 * Implementation of hook_filter_tips().
 */
function texy_filter_tips($delta, $format, $long = false) {
  return 'Texy! tips: <a href="http://texy.info/en" target="_top">http://texy.info/en</a>...';
}

/**
 * Main Texy! filter.
 */
function _filter_texy($text) {
  // get base module path
  $module_path = drupal_get_path('module', 'texy');

  // include Texy!
  require_once $module_path . '/texy/texy.php';
  $texy = &new Texy();

  // check for use syntax highlighter
  if (variable_get('texy_use_syntaxhighlight', false)) {
    $texy_syntaxhighlighter_name = variable_get('texy_syntaxhighlighter_name', '');
    // check name of syntax highlighter
    if ($texy_syntaxhighlighter_name == '') {
      // set user callback function for /-- code blocks (PHP only)
      $texy->blockModule->codeHandler = '_filter_texy_basic_highlighter';
    }
    else {
      $texy_syntaxhighlighter = $module_path . '/lib/' . $texy_syntaxhighlighter_name . '.php';
      $texy_syntaxhighlighter_dir = $module_path . '/lib/' . $texy_syntaxhighlighter_name;
      if (file_exists($texy_syntaxhighlighter) && file_exists($texy_syntaxhighlighter_dir)) {
        // set user callback function for /-- code blocks (for configured syntax highlighter)
        $texy->blockModule->codeHandler = '_filter_texy_' . $texy_syntaxhighlighter_name . '_highlighter';
        require_once $texy_syntaxhighlighter;
      }
      else {
        drupal_set_message(t('Syntax highlighter <em>' . $texy_syntaxhighlighter_name . '</em> is not ready...'),'error');
      }
    }
  }

  // other optional configuration
  $texy->utf = TRUE;                    // enable UTF-8, always on for DRUPAL
  // basic settings
  $_allowedclasses = variable_get('texy_allowed_classes', '');
  if ($_allowedclasses == '<all>') {
    // permit all classes
    $texy->allowedClasses = true;
  } else if (empty($_allowedclasses)) {
    // permit no classes
    $texy->allowedClasses = false;
  } else {
    // permit only specified classes
    $texy->allowedClasses = preg_split('/,/', str_replace(' ', '', $_allowedclasses));
  }

  $_allowedstyles = variable_get('texy_allowed_styles', '');
  if ($_allowedstyles == '<all>') {
    // permit all styles
    $texy->allowedStyles = true;
  } else if (empty($_allowedstyles)) {
    // permit no styles
    $texy->allowedStyles = false;
  } else {
    // permit only specified styles
    $texy->allowedStyles = preg_split('/,/', str_replace(' ', '', $_allowedstyles));
  }

  $texy->imageModule->root = variable_get('texy_image_dir', 'files/images');
  $texy->imageModule->rootPrefix;
  $texy->headingModule->top = variable_get('texy_heading_base', 2); // start hedings from h2

  // main Texy! filter processing
  $html = $texy->process($text);
  return $html;
}

/**
 * basic syntax highlighter (if there is no syntax highlighter name defined)
 */
function _filter_texy_basic_highlighter(&$element) {
  $lang = strtoupper($element->lang);
  // only for /---code php
  if ($lang != 'PHP') {
    return;
  }

  $element->setContent(highlight_string("<?php\n" . decode_entities($element->content) . "\n?>", TRUE), TRUE);
}
